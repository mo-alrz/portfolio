-- Rank top entities and calculate contribution to total
WITH ranked_data AS (
  SELECT category,
    SUM(value) AS total_value,
    ROW_NUMBER() OVER (PARTITION BY category ORDER BY SUM(value) DESC) AS rank,
    SUM(value) OVER () AS overall_total
  FROM dataset
  GROUP BY category
)
SELECT *,
  (total_value * 1.0 / overall_total) AS contribution_ratio
FROM ranked_data;
